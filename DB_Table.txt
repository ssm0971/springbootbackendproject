CREATE TABLE TBL_REPORT (
	REPORT_NO NUMBER PRIMARY KEY,
	REPORT_CATE VARCHAR2(100) NOT NULL CONSTRAINT REPORT_CATE_CEHECK CHECK (REPORT_CATE IN ('입양', '임시보호', '후기', '봉사', '기부', '카풀', '자랑', '커뮤')),
	REPORT_CONTENT VARCHAR2(200),
	REPORT_REGIDATE DATE DEFAULT SYSDATE,
	REPORT_WRITER NUMBER NOT NULL,
	REPORT_CONTENT_NO NUMBER NOT NULL
);
